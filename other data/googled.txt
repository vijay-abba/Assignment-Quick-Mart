    def user_exist(self):
        files = list(Path("").glob("*.txt"))
        found_user = False
        for i in files:
            file_name = str(i).split(".")[0]

            if file_name == self.username:
                print("User EXIST")
                found_user = True
        else:
            print("No user found")
            found_user = False
